cmake_minimum_required(VERSION 3.15)
project(radiation_injector)
find_package(common)
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_SOURCES xradiation_injector.c)
collect (PROJECT_LIB_SOURCES xradiation_injector_sinit.c)
collect (PROJECT_LIB_SOURCES xradiation_injector_linux.c)
collect (PROJECT_LIB_SOURCES xradiation_injector_g.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_HEADERS xradiation_injector.h)
collect (PROJECT_LIB_HEADERS xradiation_injector_hw.h)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
include_directories(${CMAKE_BINARY_DIR}/include)
add_library(radiation_injector STATIC ${_sources})
set_target_properties(radiation_injector PROPERTIES LINKER_LANGUAGE C)
